# 常用核心模块 fs （文件系统）

const fs = require('fs');

  • fs.readFile
  • fs.writeFile

# 文件读取 fs.readFile()

语法 ： fs.readFile(file-path , base-type , callback(ERR,DATA) )

  • 参数1 ： 文件路径
  • 参数2 ： 读取文件的编码格式，默认是null
  • 参数3 ： 文件读取完成的回调函数，第一个参数ERR对象,如果读取没有出错， ERR会是null， 第二个参数DATA是读取成功后返回的值
// 引入文件模块
const fs = require('fs');
fs.readFile('./name.txt', 'utf-8', (err,data)=> if err console.log(data))
//将会读取到name.txt 文件中的内容， 控制台会打印出其内容

# 文件写入 fs.writeFile()

语法 ： fs.writeFile(file-path , body|string , base-type(可选) , callback(ERR) )

  • 参数1 ： 文件路径
  • 参数2 ： 写入的数据可以给定一个字符串
  • 参数3 ： （可选参数）写入文件的编码格式，默认是utf8 ，一般该参数不写
  • 参数4 ： 文件读取完成的回调函数，只有一个参数ERR对象,如果读取没有出错， ERR会是null。
const fs = require('fs');
let body = '你好啊';
fs.writeFile('./name.txt', body, err => err || console.log('修改成功'))
//最终会将name.txt 文件中的内容修改成‘你好啊’，并在控制台打印出成功

当文件路径所指向的文件不存在， 怎么会自动创建该目标文件 ，然后再写入 。
如果目标文件存在，则会覆盖之前的文件，也就是说原文件的内容会被全部覆盖 。

# 文件追加 fs.appendFile()

语法 ： fs.appendFile(file-path , body|string , base-type(可选) , callback(ERR) )

  • 参数1 ： 文件路径
  • 参数2 ： 追加的数据可以给定一个字符串
  • 参数3 ： （可选参数）写入文件的编码格式，默认是utf8 ，一般该参数不写
  • 参数4 ： 文件读取完成的回调函数，只有一个参数ERR对象,如果读取没有出错， ERR会是null。
const fs = require('fs');
let body = 'XX';
fs.appendFile('./name.txt', body, err => err||console.log('追加成功'))
//最终会将name.txt 文件中的内容修改成‘你好啊XX’，并在控制台打印出追加成功

# 文件信息 fs.stat()

语法 ： fs.stat(file-path , callback(ERR,STAT) )

  • 参数1 ： 文件路径
  • 参数2 ： 文件读取完成的回调函数，只有一个参数ERR对象,如果读取没有出错， ERR会是null。第二个参数STAT则保存了该文件对应的信息
const fs = require('fs');
fs.stat('./name.txt', (err,stat) => err||console.log(stat))

# 文件内目录信息 fs.readdir() （异步）

语法 ： fs.readdir(file-path , callback(ERR,FILE) )

  • 参数1 ： 文件路径
  • 参数2 ： 文件读取完成的回调函数，只有一个参数ERR对象,如果读取没有出错， ERR会是null。第二个参数FILE则保存了该文件目录对应的信息
//注意该操作是异步的
const fs = require('fs');
fs.readdir('./name', (err,file) => err||console.log(file))
//最终会议数组的方式打印出结果类似如下的格式打印出来
//[ 'dc', 'idnex.html', 'index.css', 'index.js' ]

# 文件拷贝 fs.copyFile() （异步）

语法 ： fs.copyFile(src , dest , callback(ERR,FILE) )

  • 参数1 ： 文件路径
  • 参数2 ： 文件读取完成的回调函数，只有一个参数ERR对象,如果读取没有出错， ERR会是null。第二个参数FILE则保存了该文件目录对应的信息
const fs = require('fs');
fs.copyFile('源文件.txt', '目标文件.txt', (err) => {
  if (err) throw err;
  console.log('源文件已拷贝到目标文件');
});
// 默认情况下将创建或覆盖目标文件。

# 常用核心模块 path （路径）

const path = require('path');

path 模块的默认操作因 Node.js 应用程序运行所在的操作系统而异。 具体来说，当在 Windows 操作系统上运行时， path 模块将假定正在使用 Windows 风格的路径。
nodejs中文path-Windows 与 POSIX
兼容写法 ath.win32

# path.basename(path[, ext])

path.basename() 方法返回 path 的最后一部分

path.basename('/foo/bar/baz/asdf/quux.html');
// 返回: 'quux.html'

path.basename('/foo/bar/baz/asdf/quux.html', '.html');
// 返回: 'quux'

# path.dirname(path)

path.dirname() 方法返回 path 的目录名

path.dirname('/foo/bar/baz/asdf/quux');
// 返回: '/foo/bar/baz/asdf'

# path.extname(path)

//POSIX 
path.basename('C:\\temp\\myfile.html');// 返回: 'C:\\temp\\myfile.html'

//windows
path.basename('C:\\temp\\myfile.html');// 返回: 'myfile.html'

//windeows + POSIX 
path.win32.basename('C:\\temp\\myfile.html');// 返回: 'myfile.html'

# 常用的模块 module （模块标识）

  • module 是CommonJS规范中定义的一个对象， 表示当前这个具体的JS模块
  • CommonJS的核心思想是通过require方法来同步加载依赖的其他模块，通过module.exports导出需要暴露的接口

# exports/module.exports 暴露模块成员

  • Node为每个模块提供一个exports变量，指向module.exports。 等同于在每一个页面第一行存在一行代码let exports = module.exports;
// .
// ├─ utils.js
// └─ app.js
//utils.js
    let appid = '123'
    let bar = function(id){
        return `id编号： ${id}`
    }
    module.exports = {
        appid,bar
    }

    // 或者直接写
    // exports.appid = appid
    // exports.bar = bar

# require()函数用于引用某个模块

//app.js
    let utils = require('./utils') // 通过require引入utils
    console.log(utils.appid) // 12345
    console.log(utils.bar(2)) // 编号：2

# module.exports和expports的区别

  • module.exports和exports默认引用了同一个空对象
  • module.exports和exports都可以向外暴露成员
  • module.exports永远是最终的导出模块，我们可以将exports当作是module.exports的简写
  • exports类似于在每一个文件首行添加了let exports = module.exports代码，所以，我们不可以使用exports = XXX的方式

注意，上面的代码我们使用module.exports = {} 但是我们不可以使用 exports = {}，该写法会切断exports与module.exports的联系
❌exports = XXX
✔ exports.XXX = XXX 或者 module.exports

推荐使用 module.exports
